binary_search.py 
def binary_search(lista, value, left, right):
    medio = (left + right) // 2
    
    if medio > (len(lista) - 1) or right < left:
        return -1
    elif lista[medio] == value:
        return medio

    if value > lista[medio]:
        left = medio + 1
    else:
        right = medio - 1

    return binary_search(lista, value, left, right)
